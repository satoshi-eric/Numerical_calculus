# definição de função para calcular todas as equações do sistema de Lorenz
def lorenz(x, y, z, sigma=10, rho=28, beta=8/3):
    dx_dt = sigma*(y - x)
    dy_dt = rho*x - y - x*z
    dz_dt = x*y - beta*z
    # retorna os resultados das equações do sistema(dx/dt, dy/dt, dz/dt)
    return dx_dt, dy_dt, dz_dt

# criação das listas contendo os pontos do gráfico
for i in range(n):
    # desempacotamento da função que possui 3 retornos
    # dx/dt, dy/dt, dz/dt
    x, y, z = lorenz(xt[i], yt[i], zt[i])
    xt[i+1] = xt[i] + (x*dt)
    yt[i+1] = yt[i] + (y*dt)
    zt[i+1] = zt[i] + (z*dt)
